---chatbot/views.py
from django.shortcuts import render

# Create your views here.
def home(request, cn='all'):
    if cn =='all': return render(request,'chatbot/index.html')
    if cn =='ie': return render(request,'chatbot/index.ie.html')

def routes(request, p=None):
    if p =='faq': return render(request,'chatbot/FAQ.html')
    if p =='p': return render(request,'chatbot/prevention.html')
    if p =='s': return render(request,'chatbot/symptoms.html')

from django.shortcuts import render
from django.contrib import admin
from django.urls import path
from chatbot import views

urlpatterns = [
    path('', views.home),
    path('ie/', lambda x: views.home(x, cn='ie')),
    path('faq/', lambda x: views.routes(x, p='faq')),
    path('p/', lambda x: views.routes(x, p='p')),
    path('s/', lambda x: views.routes(x, p='s'))   
]
